231 public function updateRoomState(Room $room): void
232 {
233 $acquisitionSystem = $room->getAcquisitionSystem();
234 if (!$acquisitionSystem) {
235 // Pas d’AcquisitionSystem => On peut décider de faire autre chose
236 return;
237 }
238
239 // Met à jour l’AcquisitionSystem depuis le JSON
240 $this->updateAcquisitionSystemFromJson($acquisitionSystem);
241
242 // Par défaut, on considère qu’on n’a pas de data
243 $state = RoomStateEnum::NO_DATA;
244
245 // Récupérer les données capteur
246 $temperature = $acquisitionSystem->getTemperature();
247 $humidity = $acquisitionSystem->getHumidity();
248 $co2 = $acquisitionSystem->getCo2();
249
250 // Vérifier si on est en période de chauffage
251 $currentMonth = (int)(new \DateTime())->format('m');
252 // Si le mois est >= 11 OU <= 4 => période de chauffage
253 $isHeatingPeriod = ($currentMonth >= 11 || $currentMonth <= 4);
254
255 $thresholds = $this->thresholdRepository->getDefaultThresholds();
256
257 // 1. Déterminer l’état des capteurs
258 $sensorState = SensorStateEnum::LINKED; // Valeur par défaut
259 if (
260 $temperature === null && $humidity === null && $co2 === null
261 || $thresholds->isTemperatureAberrant($temperature)
262 || $thresholds->isHumidityAberrant($humidity)
263 || $thresholds->isCo2Aberrant($co2)
264 ) {
265 $sensorState = SensorStateEnum::NOT_WORKING;
266 }
267
268 // 2. Évaluer la température
269 if ($temperature !== null && !$thresholds->isTemperatureAberrant($temperature)) {
270 if ($isHeatingPeriod) {
271 if (
272 $temperature < $thresholds->getHeatingTempCriticalMin() ||
273 $temperature > $thresholds->getHeatingTempCriticalMax()
274 ) {
275 $state = RoomStateEnum::CRITICAL;
276 } elseif (
277 $temperature < $thresholds->getHeatingTempWarningMin() ||
278 $temperature > $thresholds->getHeatingTempWarningMax()
279 ) {
280 // Ne passer en AT_RISK que si pas déjà CRITICAL
281 if ($state !== RoomStateEnum::CRITICAL) {
282 $state = RoomStateEnum::AT_RISK;
283 }
284 }
285 } else {
286 // Non-heating period
287 if (
288 $temperature < $thresholds->getNonHeatingTempCriticalMin() ||
289 $temperature > $thresholds->getNonHeatingTempCriticalMax()
290 ) {
291 $state = RoomStateEnum::CRITICAL;
292 } elseif (
293 $temperature < $thresholds->getNonHeatingTempWarningMin() ||
294 $temperature > $thresholds->getNonHeatingTempWarningMax()
295 ) {
296 if ($state !== RoomStateEnum::CRITICAL) {
297 $state = RoomStateEnum::AT_RISK;
298 }
299 }
300 }
301 }
302
303 // 3. Évaluer le CO2
304 if ($co2 !== null && !$thresholds->isCo2Aberrant($co2)) {
305 if ($co2 < $thresholds->getCo2CriticalMin() || $co2 > $thresholds->getCo2ErrorMax()) {
306 $state = RoomStateEnum::CRITICAL;
307 } elseif ($co2 > $thresholds->getCo2WarningMin() && $co2 <= $thresholds->getCo2CriticalMax()) {
308 if ($state !== RoomStateEnum::CRITICAL) {
309 $state = RoomStateEnum::AT_RISK;
310 }
311 }
312 }
313
314 // 4. Évaluer l’humidité
315 if ($humidity !== null && !$thresholds->isHumidityAberrant($humidity)) {
316 if ($humidity < $thresholds->getHumCriticalMin()) {
317 $state = RoomStateEnum::CRITICAL;
318 } elseif (
319 $humidity < $thresholds->getHumWarningMin() ||
320 ($humidity > $thresholds->getHumWarningMax() && $humidity <= $thresholds->getHumCriticalMax())
321 ) {
322 if ($state !== RoomStateEnum::CRITICAL) {
323 $state = RoomStateEnum::AT_RISK;
324 }
325 } elseif ($humidity > $thresholds->getHumCriticalMax()) {
326 $state = RoomStateEnum::CRITICAL;
327 }
328 }
329
330 if (
331 $state === RoomStateEnum::NO_DATA &&
332 ($temperature !== null || $humidity !== null || $co2 !== null)
333 ) {
334 $state = RoomStateEnum::STABLE;
335 }
336
337 // 5. Créer une tâche de maintenance si capteur KO et qu’on n’est pas dans NO_DATA
338 if ($sensorState === SensorStateEnum::NOT_WORKING && $state !== RoomStateEnum::NO_DATA) {
339 $this->createTaskForTechnician($room);
340 }
341
342 // 6. Mise à jour finale du Room et de l’AcquisitionSystem
343 $utcDateTime = new \DateTime('now', new \DateTimeZone('Europe/Paris'));
344 $room->setLastUpdatedAt($utcDateTime);
345 $room->setSensorState($sensorState);
346 $acquisitionSystem->setState($sensorState);
347 $room->setState($state);
348
349 $em = $this->getEntityManager();
350 $em->persist($room);
351 $em->flush();
352 }

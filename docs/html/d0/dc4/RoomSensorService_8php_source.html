129 if (isset($entry['dateCapture'])) {
130 try {
131 $captureDate = new \DateTime($entry['dateCapture']);
132 if ($lastCapturedAt === null || $captureDate > $lastCapturedAt) {
133 $lastCapturedAt = $captureDate;

136 throw new \RuntimeException(
137 "Invalid date format in entry: " . json_encode($entry) . ". Error: " . $e->getMessage()

142 if ($lastCapturedAt !== null) {
143 $acquisitionSystem->setLastCapturedAt($lastCapturedAt);
144 }

147 $em->persist($acquisitionSystem);
148 $em->flush();

208 public function updateRoomState(Room $room): void
209 {
210 $acquisitionSystem = $room->getAcquisitionSystem();

213 }
214
215 // 1. Update the AcquisitionSystem directly from the API
216 $this->updateAcquisitionSystemFromApi($acquisitionSystem);
217
218 // 2. Default room state
219 $state = RoomStateEnum::NO_DATA;
220
221 // 3. Retrieve sensor values
222 $temperature = $acquisitionSystem->getTemperature();
223 $humidity = $acquisitionSystem->getHumidity();
224 $co2 = $acquisitionSystem->getCo2();
225
226 // 4. Determine if we are in heating period
227 $currentMonth = (int)(new \DateTime())->format('m');
228 $isHeatingPeriod = ($currentMonth >= 11 || $currentMonth <= 4);
229
230 $thresholds = $this->thresholdRepository->getDefaultThresholds();
231
232 // 5. Determine sensor state
233 $sensorState = SensorStateEnum::LINKED;
234 if (
235 ($temperature === null && $humidity === null && $co2 === null)
236 || $thresholds->isTemperatureAberrant($temperature)
237 || $thresholds->isHumidityAberrant($humidity)
238 || $thresholds->isCo2Aberrant($co2)
239 ) {
240 $sensorState = SensorStateEnum::NOT_WORKING;
241 }
242
243 // 6. Evaluate temperature
244 if ($temperature !== null && !$thresholds->isTemperatureAberrant($temperature)) {
245 if ($isHeatingPeriod) {
246 // Heating period
247 if (
248 $temperature < $thresholds->getHeatingTempCriticalMin() ||
249 $temperature > $thresholds->getHeatingTempCriticalMax()
250 ) {
251 $state = RoomStateEnum::CRITICAL;
252 } elseif (
253 $temperature < $thresholds->getHeatingTempWarningMin() ||
254 $temperature > $thresholds->getHeatingTempWarningMax()
255 ) {
256 if ($state !== RoomStateEnum::CRITICAL) {
257 $state = RoomStateEnum::AT_RISK;
258 }
259 }
260 } else {
261 // Non-heating period
262 if (
263 $temperature < $thresholds->getNonHeatingTempCriticalMin() ||
264 $temperature > $thresholds->getNonHeatingTempCriticalMax()
265 ) {
266 $state = RoomStateEnum::CRITICAL;
267 } elseif (
268 $temperature < $thresholds->getNonHeatingTempWarningMin() ||
269 $temperature > $thresholds->getNonHeatingTempWarningMax()
270 ) {
271 if ($state !== RoomStateEnum::CRITICAL) {
272 $state = RoomStateEnum::AT_RISK;
273 }
274 }
275 }
276 }
277
278 // 7. Evaluate CO2
279 if ($co2 !== null && !$thresholds->isCo2Aberrant($co2)) {
280 if ($co2 < $thresholds->getCo2CriticalMin() || $co2 > $thresholds->getCo2ErrorMax()) {
281 $state = RoomStateEnum::CRITICAL;
282 } elseif ($co2 > $thresholds->getCo2WarningMin() && $co2 <= $thresholds->getCo2CriticalMax()) {
283 if ($state !== RoomStateEnum::CRITICAL) {
284 $state = RoomStateEnum::AT_RISK;
285 }
286 }
287 }
288
289 // 8. Evaluate humidity
290 if ($humidity !== null && !$thresholds->isHumidityAberrant($humidity)) {
291 if ($humidity < $thresholds->getHumCriticalMin()) {
292 $state = RoomStateEnum::CRITICAL;
293 } elseif (
294 $humidity < $thresholds->getHumWarningMin()
295 || ($humidity > $thresholds->getHumWarningMax() && $humidity <= $thresholds->getHumCriticalMax())
296 ) {
297 if ($state !== RoomStateEnum::CRITICAL) {
298 $state = RoomStateEnum::AT_RISK;
299 }
300 } elseif ($humidity > $thresholds->getHumCriticalMax()) {
301 $state = RoomStateEnum::CRITICAL;
302 }
303 }
304
305 // If we have sensor data but haven't triggered any risk, set to STABLE
306 if (
307 $state === RoomStateEnum::NO_DATA &&
308 ($temperature !== null || $humidity !== null || $co2 !== null)
309 ) {
310 $state = RoomStateEnum::STABLE;
311 }
312
313 // 9. If the sensor is NOT_WORKING but we do have data, create a maintenance task
314 if ($sensorState === SensorStateEnum::NOT_WORKING && $state !== RoomStateEnum::NO_DATA) {
315 $this->createTaskForTechnician($room);
316 }
317
318 // 10. Update Room and AcquisitionSystem
319 $utcDateTime = new \DateTime('now', new \DateTimeZone('Europe/Paris'));
320 $room->setLastUpdatedAt($utcDateTime);
321 $room->setSensorState($sensorState);
322 $acquisitionSystem->setState($sensorState);
323 $room->setState($state);
324
325 $em = $this->getEntityManager();
326 $em->persist($room);
327 $em->flush();

203 return $this->redirectToRoute('app_todolist');
204 }
205
206 if ($action->getInfo() == ActionInfoEnum::ASSIGNMENT) {
207 $acquisitionSystemId = $request->request->get('acquisitionSystem');
208 $acquisitionSystem = $acquisitionSystemRepository->find($acquisitionSystemId);

210 if (!$acquisitionSystem) {

212 return $this->redirectToRoute('app_todolist');
213 }
214
215 $room = $action->getRoom();

217 $room->setSensorState(SensorStateEnum::LINKED);
218 $room->setAcquisitionSystem($acquisitionSystem);

220 $action->setAcquisitionSystem($acquisitionSystem);
221 }
222
223 if ($action->getInfo() == ActionInfoEnum::MAINTENANCE) {
224 $maintenanceStatus = $request->request->get('maintenanceStatus');
225
226 if ($maintenanceStatus === 'yes') {
227 // If the system is repaired, update its state
228 $acquisitionSystem = $action->getAcquisitionSystem();

231 $entityManager->persist($acquisitionSystem);

234 } else {
235 $this->addFlash('warning', 'Please select a valid maintenance status.');
236 return $this->redirectToRoute('app_todolist_edit', ['id' => $action->getId()]);
237 }
238 }
239
240
241 if ($action->getInfo() == ActionInfoEnum::UNASSIGNMENT) {
242 $room = $action->getRoom();
243
244 if ($room) {

254

256
257 $managers = $userRepository->findByRole('ROLE_MANAGER');
258
259 // Ajout de la vérification pour s'assurer qu'il y a au moins un manager

262 if ($managerUser) {

265 }

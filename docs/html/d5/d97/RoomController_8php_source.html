87 date_default_timezone_set('Europe/Paris');
88 // Récupération de toutes les salles disposant d'un AcquisitionSystem.
89 $rooms = $roomRepository->findRoomsWithAS();
90
91 // --------------------------------------
92 // Vérification de la fraîcheur de lastUpdatedAt
93 // et mise à jour si nécessaire (seulement si plus vieux que 3 minutes)
94 // --------------------------------------
95 foreach ($rooms as $room) {
96 // On ne traite que les salles vraiment équipées (LINKED)
97 if ($room->getSensorState() === SensorStateEnum::LINKED) {
98 $lastUpdatedAt = $room->getLastUpdatedAt();
99 $now = new \DateTimeImmutable('now');
100
101 // Si lastUpdatedAt est nul ou date de plus de 3 minutes
102 if (
103 null === $lastUpdatedAt
104 || ($now->getTimestamp() - $lastUpdatedAt->getTimestamp()) >= 180
105 ) {
106 // => on met à jour la salle (updateRoomState)
107 try {
108 $this->roomSensorService->updateRoomState($room);
109 } catch (\Exception $e) {
110 // Gérer ou logger l’erreur
111 continue;
112 }
113 } else {
114 // Sinon, on ne fait qu'un "load" local pour ne pas surcharger
115 try {
116 $this->roomSensorService->loadSensorData($room->getAcquisitionSystem());
117 } catch (\Exception $e) {
118 // Gérer ou logger l’erreur
119 }
120 }
121 }
122 }
123
124 $stateParam = $request->query->get('state');
125 $stateEnum = $stateParam ? RoomStateEnum::tryFrom($stateParam) : null;
126
127 // Formulaire de filtre
128 $filterForm = $this->createForm(FilterRoomType::class, null, [
129 'state' => $stateEnum,
130 ]);
131 $filterForm->handleRequest($request);
132
133 $criteria = [];
134
135 // Filtering pour utilisateurs non authentifiés
136 if (!$this->getUser()) {
137 $criteria['sensorStatus'] = 'linked';
138 $criteria['state_not'] = 'no data'; // Exclure les états "no data"
139 }
140
141 // Filtrage initial (via URL)
142 if ($stateParam) {
143 $criteria['state'] = $stateParam;
144 }
145
146 // Reset du formulaire
147 if ($filterForm->get('reset')->isClicked()) {
148 return $this->redirectToRoute('app_rooms');
149 }
150
151 // Application des critères du formulaire
152 if ($filterForm->isSubmitted() && $filterForm->isValid()) {
154 $data = $filterForm->getData();
155
156 if (!empty($data->getName())) {
157 $criteria['name'] = $data->getName();
158 }
159 if ($data->getFloor()) {
160 $criteria['floor'] = $data->getFloor();
161 }
162 if ($data->getState()) {
163 $criteria['state'] = $data->getState();
164 }
165 }
166
167 // Récupération des salles selon critères
168 $rooms = $roomRepository->findByCriteria($criteria);
169
170 // Récupération des actions en cours pour chaque salle
171 $ongoingTasksByRoomId = [];
172 foreach ($rooms as $room) {
173 $ongoingTask = $actionRepository->findOngoingTaskForRoom($room->getId());
174 $ongoingTasksByRoomId[$room->getId()] = $ongoingTask;
175 }
176
177 return $this->render('rooms/index.html.twig', [
178 'rooms' => $rooms,
179 'filterForm' => $filterForm->createView(),
180 'ongoingTasks' => $ongoingTasksByRoomId,
181 ]);
